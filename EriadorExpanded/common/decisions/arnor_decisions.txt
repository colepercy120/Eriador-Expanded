
decision_integrate_the_southern_provinces = {
	title = decision_integrate_the_southern_provinces
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_rhudaur_02.dds"
	}
	
	selection_tooltip = decision_integrate_the_southern_provinces_tooltip
	desc = decision_integrate_the_southern_provinces_desc
	confirm_text = decision_integrate_the_southern_provinces_confirm

	decision_group_type = realms_in_exile_major
	
	ai_goal = yes
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 100
	}

	is_shown = {
		is_independent_ruler = yes
		
		lotr_has_numenorean_culture_trigger = yes
		
		OR = {
			has_title = title:e_arnor
			AND = {
				has_title = title:e_reunitedkingdom
				is_target_in_global_variable_list = {
					name = unavailable_unique_decisions
					target = flag:decision_solidify_reunited_kingdom
				}
			}
			AND = {
				has_title = title:e_numenor_in_exile
				is_target_in_global_variable_list = {
					name = unavailable_unique_decisions
					target = flag:decision_numenor_in_exile
				}
			}
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:decision_integrate_the_southern_provinces
			}
		}
	}

	is_valid = {
        completely_controls = title:k_dunland
        completely_controls = title:k_enedhwaith
    }

    is_valid_showing_failures_only = {
        is_imprisoned = no
        is_available_adult = yes
        is_at_war = no
    }

	effect = {
		trigger_event = {
			id = ee_arnor.1
			days = 1
		}
		dynasty = { add_dynasty_prestige = 500 }
		hidden_effect = {
			if = {
				limit = {
					has_title = title:e_reunitedkingdom
					is_target_in_global_variable_list = {
						name = unavailable_unique_decisions
						target = flag:decision_solidify_reunited_kingdom
					}
				}
				title:k_dunland = { set_de_jure_liege_title = title:e_reunitedkingdom }
				title:k_enedhwaith = { set_de_jure_liege_title = title:e_reunitedkingdom }
			}
			else_if = {
				limit = { has_title = title:e_numenor_in_exile }
				title:k_dunland = { set_de_jure_liege_title = title:e_numenor_in_exile }
				title:k_enedhwaith = { set_de_jure_liege_title = title:e_numenor_in_exile }
			}
			else = {
				title:k_dunland = { set_de_jure_liege_title = title:e_arnor }
				title:k_enedhwaith = { set_de_jure_liege_title = title:e_arnor }
			}	
		}	
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_integrate_the_southern_provinces
		}
	}	
}
